## 一维数组去重

### ES6 new Set去重
``` bash
    function fn (arr) {
        return Array.from(new Set(arr));
    }
```
### for循环嵌套去重
``` bash
    function fn (arr) {
        for ( var i = 0; i < arr.length; i++ ) {
            for ( var j=i+1; j<arr.length; j++ ) {
                if ( arr[i] == arr[j] ) {
                    arr.splice(j,1);
                    j--;
                }
            }
        }
        return arr;
    }
```
### 使用indexOf去重
``` bash
    function fn (arr) {
        let array = [];
        for ( let i = 0; i < arr.length; i++ ) {
            if ( array.indexOf(arr[i]) === -1 ) {
                array.push(arr[i])
            }
        }
        return array;
    }
```
### 利用对象属性唯一性去重
``` bash
    function fn (arr) {
        let array = [],obj = {};
        for ( let i = 0; i < arr.length; i++ ) {
            if ( !obj[arr[i]] ) {
                array.push(arr[i]);
                obj[arr[i]] = 1;
            } else {
                obj[arr[i]]++
            }
        }
        return array;
    }
```

### 两个简单数组求差异
``` bash
    var arr1 = [1,2,3,4,5],arr2 = [1,2,5,6,7];
    function arrayDifferences(arr1,arr2) {
        var obj1 = new Set(arr1);
        var obj2 = new Set(arr2);

        var subset = [];
        for ( let key of obj1 ) {
            if ( !obj2.has(key) ) {
                subset.push(key);
            }
        }
        return subset;
    }
```
